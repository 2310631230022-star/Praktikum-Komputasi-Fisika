# SESI 1A: Implementasi Manual Tiga Metode Beda Hingga
import numpy as np
import matplotlib.pyplot as plt

# Definisi fungsi dan turunan analitik
def f(x):
    return x**3 + 2*x**2 - 3*x + 1
def df_eksak(x): return 3*x**2 + 4*x - 3
def d2f_eksak(x): return 6*x + 4

# Tiga metode beda hingga
def beda_maju(f, x, h):
    return (f(x+h) - f(x)) / h
def beda_mundur(f, x, h):
    return (f(x) - f(x-h)) / h
def beda_pusat(f, x, h):
    return (f(x+h) - f(x-h)) / (2*h)
def beda_kedua(f, x, h):
    return (f(x+h) - 2*f(x) + f(x-h)) / h**2

# SESI 2: Regresi Polinomial berbasis Normal Equations Matriks
# ============================================================
def regres_polinomial(x, y, m):
    '''
    Regresi polinomial orde m dari Normal Equations.
    Parameter:
    x, y : array data
    m    : orde polinomial
    Return:
    a      : koefisien [a0, a1, ..., am]
    R2     : koefisien determinasi
    SS_res : jumlah kuadrat residual
    '''
    n = len(x)
    # Bangun matriks desain (Vandermonde) X berukuran n×(m+1)
    X = np.column_stack([x**k for k in range(m+1)])
    
    # Normal Equations: (XᵀX)·a = Xᵀy
    XtX = X.T @ X
    Xty = X.T @ y
    
    # Selesaikan sistem linear
    a = np.linalg.solve(XtX, Xty)
    
    # Hitung R²
    y_pred = X @ a
    SS_res = np.sum((y - y_pred)**2)
    SS_tot = np.sum((y - y.mean())**2)
    R2     = 1 - SS_res / SS_tot
    
    return a, R2, SS_res

# ─── BANDINGKAN BERBAGAI ORDE ─────────────────────────────
print('=== PERBANDINGAN ORDE POLINOMIAL ===')
print(f'{"Orde":>5} {"R²":>10} {"SS_res":>12} {"Koefisien utama"}')
print('-' * 70)

orde_list = [1, 2, 3, 4, 5, 7, 10]
hasil = {}

for m in orde_list:
    a_m, R2_m, SSr_m = regres_polinomial(x, y, m)
    hasil[m] = (a_m, R2_m, SSr_m)
    
    a_str = ', '.join([f'{c:.3f}' for c in a_m[:4]])
    if len(a_m) > 4:
        a_str += ', ...'
    print(f'{m:>5} {R2_m:>10.6f} {SSr_m:>12.4f} [{a_str}]')

print('\nPerhatikan: R² terus naik tapi apakah model makin baik?')

# ─── VISUALISASI: FITTING BERBAGAI ORDE ──────────────────
fig, axes = plt.subplots(2, 3, figsize=(16, 10))
axes = axes.flatten()

orde_vis = [1, 2, 3, 4, 5, 10]
x_fit = np.linspace(x.min()-0.3, x.max()+0.3, 400)

for idx, m in enumerate(orde_vis):
    ax = axes[idx]
    a_m, R2_m, SSr_m = hasil[m]
    
    # Evaluasi polinomial
    y_fit = np.polyval(a_m[::-1], x_fit) # polyval butuh koef descending
    
    ax.scatter(x, y, color='steelblue', s=50, zorder=5)
    ax.plot(x_fit, y_fit, 'r-', lw=1.8)
    
    ax.set_xlim(-0.5, 10.5); ax.set_ylim(-5, 60)
    ax.set_title(f'Orde {m} (R²={R2_m:.4f})', fontsize=11)
    ax.set_xlabel('x'); ax.set_ylabel('y')
    
    # Tandai jika overfitting (osilasi besar di tepi)
    y_edge = np.polyval(a_m[::-1], np.array([-0.3, 10.3]))
    if np.any(np.abs(y_edge) > 60):
        ax.set_facecolor('#451a1e')
        ax.set_title(f'Orde {m} (R²={R2_m:.4f})  ⚠️  OVERFIT', fontsize=10, color='#f38ba8')

plt.suptitle('Regresi Polinomial: Berbagai Orde', fontsize=13, y=1.01, color='#cdd6f4')

# ============================================================
# SESI 3: Hukum Pendinginan Newton — Linearisasi Eksponensial
# Model: T(t) = T∞ + (T₀−T∞)·e^(-k·t)
# Setelah transformasi: ln(T(t)−T∞) = ln(T₀−T∞) − k·t
# Bentuk linear: Y = A + B·t, di mana Y = ln(T−T∞)
#
# Data simulasi: Bola besi panas didinginkan di udara
# ============================================================
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import curve_fit

# ─── DATA PENGAMATAN ──────────────────────────────────────
# Waktu (menit) dan suhu (°C)
t_data = np.array([0, 2, 4, 6, 8, 10, 12, 15, 20, 25, 30], dtype=float)
T_data = np.array([90.0, 78.5, 68.8, 60.2, 53.4, 47.8, 43.1, 37.2, 29.6, 24.8, 22.1], dtype=float)
T_inf = 21.0 # suhu lingkungan (°C)

print('=== DATA PENDINGINAN NEWTON ===')
print(f'Suhu lingkungan T∞ = {T_inf}°C')
print(f't (menit): {t_data}')
print(f'T (°C)   : {T_data}')

# ─── LINEARISASI: Y = ln(T − T∞) ─────────────────────────
dT = T_data - T_inf # kelebihan suhu di atas T∞
Y  = np.log(dT)     # variabel transformasi

print(f'\nT − T∞ : {np.round(dT, 2)}')
print(f'ln(T−T∞): {np.round(Y, 4)}')

# ─── REGRESI LINEAR pada (t, Y) ───────────────────────────
n     = len(t_data)
x_bar = np.mean(t_data)
Y_bar = np.mean(Y)

b_lin = np.sum((t_data - x_bar)*(Y - Y_bar)) / np.sum((t_data - x_bar)**2)
a_lin = Y_bar - b_lin * x_bar

# Kembali ke parameter fisika
k_lin  = -b_lin                 # laju pendinginan [1/menit]
T0_lin = np.exp(a_lin) + T_inf  # suhu awal terfitting

Y_pred = a_lin + b_lin * t_data
R2_lin = 1 - np.sum((Y - Y_pred)**2) / np.sum((Y - Y_bar)**2)

print(f'\n=== HASIL LINEARISASI ===')
print(f'Intersep A = ln(T₀−T∞) = {a_lin:.4f}')
print(f'Kemiringan B = −k      = {b_lin:.4f}')
print(f'k (laju pendinginan)   = {k_lin:.4f} /menit')
print(f'T₀ terfitting          = {T0_lin:.2f}°C (data: {T_data[0]}°C)')
print(f'R² (pada Y=ln(T−T∞))   = {R2_lin:.6f}')

# ─── VERIFIKASI: curve_fit langsung di domain y asli ────
def model_newton(t, T0, k):
    return T_inf + (T0 - T_inf) * np.exp(-k * t)

popt, pcov = curve_fit(model_newton, t_data, T_data, p0=[90.0, 0.1], maxfev=5000)
T0_cf, k_cf = popt
perr = np.sqrt(np.diag(pcov)) # standard error

T_pred_cf = model_newton(t_data, *popt)
SS_res_cf = np.sum((T_data - T_pred_cf)**2)
SS_tot_cf = np.sum((T_data - T_data.mean())**2)
R2_cf     = 1 - SS_res_cf / SS_tot_cf

print(f'\n=== VERIFIKASI scipy.optimize.curve_fit ===')
print(f'k  = {k_cf:.4f} ± {perr[1]:.4f} /menit')
print(f'T₀ = {T0_cf:.2f} ± {perr[0]:.2f} °C')
print(f'R² (pada T asli) = {R2_cf:.6f}')
print(f'\nSelisih k: {abs(k_lin-k_cf):.4f} /menit')
print('→ Linearisasi menghasilkan koefisien sedikit berbeda karena')
print(' meminimalkan residual pada ln(T) bukan pada T!')

# ─── VISUALISASI PENDINGINAN NEWTON ──────────────────────
t_fit = np.linspace(0, 35, 300)
fig, axes = plt.subplots(1, 3, figsize=(16, 5))

# Panel 1: T vs t (domain asli)
axes[0].scatter(t_data, T_data, s=70, color='steelblue', zorder=5, label='Data')
T_pred_lin = T_inf + np.exp(a_lin + b_lin * t_fit)
axes[0].plot(t_fit, T_pred_lin, 'r--', lw=2, label=f'Linearisasi (k={k_lin:.3f})')
axes[0].plot(t_fit, model_newton(t_fit, *popt), 'g-', lw=2, label=f'curve_fit (k={k_cf:.3f})')
axes[0].axhline(T_inf, color='gray', ls=':', lw=1, label=f'T∞={T_inf}°C')
axes[0].set_xlabel('Waktu (menit)'); axes[0].set_ylabel('Suhu (°C)')
axes[0].set_title('Hukum Pendinginan Newton'); axes[0].legend(fontsize=9)

# Panel 2: ln(T−T∞) vs t (domain linear)
axes[1].scatter(t_data, Y, s=70, color='darkorange', zorder=5, label='ln(T−T∞)')
axes[1].plot(t_fit, a_lin + b_lin*t_fit, 'r-', lw=2, label=f'Regresi linear\nR²={R2_lin:.4f}')
axes[1].set_xlabel('Waktu (menit)'); axes[1].set_ylabel('ln(T − T∞)')
axes[1].set_title('Domain Terliniearisasi'); axes[1].legend(fontsize=9)

# Panel 3: Residual perbandingan
res_lin = T_data - (T_inf + np.exp(a_lin + b_lin * t_data))
res_cf  = T_data - model_newton(t_data, *popt)

axes[2].plot(t_data, res_lin, 'r-o', ms=5, lw=1.5, label='Linearisasi')
axes[2].plot(t_data, res_cf, 'g-s', ms=5, lw=1.5, label='curve_fit')
axes[2].axhline(0, color='black', lw=1)
axes[2].set_xlabel('Waktu (menit)'); axes[2].set_ylabel('Residual (°C)')
axes[2].set_title('Perbandingan Residual'); axes[2].legend()

plt.tight_layout()
plt.savefig('pendinginan_newton.png', dpi=150, bbox_inches='tight')
plt.show()

t_half = np.log(2) / k_cf
print(f'\nWaktu paruh pendinginan: t½ = ln(2)/k = {t_half:.2f} menit')

# ============================================================
# SESI 4: Hukum Stefan-Boltzmann — P = ε·σ·A·T⁴
#
# Hukum asli: P = C·T^n (hukum pangkat / power law)
# Linearisasi log-log: log(P) = log(C) + n·log(T)
#
# Dari fisika: n harus = 4, C = ε·σ·A
# Kita akan memverifikasi ini secara numerik dari data!
# ============================================================
import numpy as np

from scipy.optimize import curve_fit

A_obj = 0.01    # luas permukaan [m²]
noise = 0.03    # 3% noise relatif

P_true = eps * sigma * A_obj * T_true**4
P_data = P_true * (1 + noise * np.random.randn(len(T_true)))

print('=== DATA RADIASI TERMAL ===')
print(f'{"T (K)":>8} {"P_true (W)":>12} {"P_data (W)":>12} {"Noise%":>7}')
for i in range(len(T_true)):
    pct = (P_data[i]-P_true[i])/P_true[i]*100
    print(f'{T_true[i]:>8.0f} {P_true[i]:>12.4e} {P_data[i]:>12.4e} {pct:>7.2f}%')

# ─── REGRESI LOG-LOG ──────────────────────────────────────
log_T = np.log10(T_true) # gunakan T_true sebagai sumbu x
log_P = np.log10(P_data)

n_d = len(T_true)

yb  = np.mean(log_P)

n_exp = np.sum((log_T-xb)*(log_P-yb)) / np.sum((log_T-xb)**2)
log_C = yb - n_exp * xb

# Ekstrak emisivitas dari C_fit = ε·σ·A
eps_fit = C_fit / (sigma * A_obj)

print(f'C_fit                  : {C_fit:.4e} (true: {eps*sigma*A_obj:.4e})')
print(f'ε_fit (dari C)         : {eps_fit:.4f} (true: {eps:.2f})')
print(f'R² (pada log-log)      : {R2_loglog:.6f}')
print(f'\nVerifikasi: eksponen = {n_exp:.2f} (fisika: 4.00)', f' selisih: {abs(n_exp-4)/4*100:.2f}%')

# ─── VISUALISASI STEFAN-BOLTZMANN ─────────────────────────
fig, axes = plt.subplots(1, 3, figsize=(16, 5))
T_fit = np.linspace(250, 2200, 500)
P_fit = C_fit * T_fit**n_exp

# Panel 1: P vs T (domain asli — kurva)
axes[0].scatter(T_true, P_data, s=70, color='steelblue', zorder=5, label='Data')
axes[0].plot(T_fit, P_fit, 'r-', lw=2, label=f'P = C·T^{n_exp:.2f}')

axes[0].set_ylabel('Daya Radiasi P (W)')
axes[0].set_title('Hukum Stefan-Boltzmann')

axes[1].scatter(log_T, log_P, s=70, color='darkorange', zorder=5, label='log(data)')
axes[1].plot(log_T, log_P_pred, 'r-', lw=2, label=f'Regresi log-log\nn={n_exp:.3f}, R²={R2_loglog:.4f}')
axes[1].set_xlabel('log₁₀(T)'); axes[1].set_ylabel('log₁₀(P)')
axes[1].set_title('Domain Log-Log (linear!)')
axes[1].legend(fontsize=9)

# Panel 3: Residual relatif
P_pred_data = C_fit * T_true**n_exp
res_rel = (P_data - P_pred_data) / P_pred_data * 100

axes[2].bar(range(len(T_true)), res_rel, color=['steelblue' if r>=0 else 'tomato' for r in res_rel], edgecolor='white')

axes[2].set_title('Residual Relatif terhadap Fitting')

# SESI 1A: Implementasi Manual Tiga Metode Beda Hingga
import numpy as np
import matplotlib.pyplot as plt

# Definisi fungsi dan turunan analitik
def f(x):
    return x**3 + 2*x**2 - 3*x + 1
def df_eksak(x): return 3*x**2 + 4*x - 3
def d2f_eksak(x): return 6*x + 4

# Tiga metode beda hingga
def beda_maju(f, x, h):
    return (f(x+h) - f(x)) / h
def beda_mundur(f, x, h):
    return (f(x) - f(x-h)) / h
def beda_pusat(f, x, h):
    return (f(x+h) - f(x-h)) / (2*h)
def beda_kedua(f, x, h):
    return (f(x+h) - 2*f(x) + f(x-h)) / h**2

# Titik evaluasi dan nilai eksak
x0 = 2.0
f_eksak_1 = df_eksak(x0)
f_eksak_2 = d2f_eksak(x0)

print(f'=== FUNGSI: f(x) = x^3 + 2x^2 - 3x + 1 di x={x0} ===')

# SESI 2: Formula 5-Titik dan Richardson Extrapolation

def beda_5titik(f, x, h):
    # Formula 5-titik untuk turunan pertama - galat O(h^4)
    return (-f(x+2*h) + 8*f(x+h) - 8*f(x-h) + f(x-2*h)) / (12*h)

def richardson(f, x, h):
    # Richardson Extrapolation: O(h^2) -> O(h^4)
    D_h = beda_pusat(f, x, h)
    D_h2 = beda_pusat(f, x, h/2)
    return (4*D_h2 - D_h) / 3

# Perbandingan semua metode
print('=== PERBANDINGAN SEMUA METODE (f di x=2) ===')
print(f"{'h':>8} {'Pusat O(h²)':>14} {'5-titik O(h⁴)':>14} {'Richardson O(h⁴)':>18}")
print('-' * 60)

h_test = [0.5, 0.1, 0.05, 0.01, 0.001]
for h in h_test:
    ep = abs(beda_pusat(f, x0, h) - f_eksak_1)
    e5 = abs(beda_5titik(f, x0, h) - f_eksak_1)
    er = abs(richardson(f, x0, h) - f_eksak_1)
    print(f'{h:>8.3f} {ep:>14.2e} {e5:>14.2e} {er:>18.2e}')

# Grafik konvergensi semua metode
harr2 = np.logspace(-6, 0, 100)
e_pusat2 = [max(abs(beda_pusat(f, x0, h) - f_eksak_1), 1e-17) for h in harr2]
e_5pt = [max(abs(beda_5titik(f, x0, h) - f_eksak_1), 1e-17) for h in harr2]
e_rich = [max(abs(richardson(f, x0, h) - f_eksak_1), 1e-17) for h in harr2]

fig, ax = plt.subplots(figsize=(9, 6))
ax.loglog(harr2, e_pusat2, 'g-', lw=1.8, label='Selisih Pusat O(h²)')

# SESI 3: Turunan Numerik pada Data Diskret (Tanpa Fungsi Analitik)
import numpy as np
import matplotlib.pyplot as plt

# DATA PENGUKURAN (simulasi + noise realistis)
g = 9.81 # m/s²
v_terminal = 30.0 # m/s
t_data = np.arange(0, 5.1, 0.2) # setiap 0.2 detik

s_true = (v_terminal**2/g) * np.log(np.cosh(g*t_data/v_terminal))
np.random.seed(77)
noise = np.random.normal(0, 0.05, len(t_data)) # noise ~5 cm
s_data = s_true + noise

print('=== DATA POSISI GERAK JATUH BEBAS ===')
print(f'Kecepatan terminal v_T = {v_terminal} m/s')
print(f'Jumlah titik data: {len(t_data)}')
print(f'Interval waktu: dt = {t_data[1]-t_data[0]:.1f} s\n')

print(f"{'t (s)':>6} {'s_data (m)':>12} {'s_true (m)':>12}")
for i in range(0, len(t_data), 3):
    print(f'{t_data[i]:>6.1f} {s_data[i]:>12.4f} {s_true[i]:>12.4f}')

# HITUNG KECEPATAN: v = ds/dt
# Metode 1: numpy.gradient
v_numpy = np.gradient(s_data, t_data)

# Metode 2: manual
dt = t_data[1] - t_data[0]
v_manual = np.zeros_like(s_data)
v_manual[0] = (-3*s_data[0] + 4*s_data[1] - s_data[2]) / (2*dt) # kiri
v_manual[-1] = (s_data[-3] - 4*s_data[-2] + 3*s_data[-1]) / (2*dt) # kanan
v_manual[1:-1] = (s_data[2:] - s_data[:-2]) / (2*dt) # interior

v_eksak = v_terminal * np.tanh(g*t_data / v_terminal)

# HITUNG PERCEPATAN: a = dv/dt = d²s/dt²
a_numpy = np.gradient(v_numpy, t_data)
a_eksak = g / np.cosh(g*t_data/v_terminal)**2

print(f"\n{'t':>5} {'v_numpy':>10} {'v_manual':>10} {'v_eksak':>10} {'a_numpy':>10} {'a_eksak':>10}")
print('-'*65)
for i in range(len(t_data)):
    print(f'{t_data[i]:>5.1f} {v_numpy[i]:>10.4f} {v_manual[i]:>10.4f} {v_eksak[i]:>10.4f} {a_numpy[i]:>10.4f} {a_eksak[i]:>10.4f}')

# VISUALISASI KINEMATIKA
fig, axes = plt.subplots(1, 3, figsize=(16, 5))
t_fit = np.linspace(0, 5, 300)
s_fit = (v_terminal**2/g)*np.log(np.cosh(g*t_fit/v_terminal))
v_fit = v_terminal*np.tanh(g*t_fit/v_terminal)
a_fit = g/np.cosh(g*t_fit/v_terminal)**2

# Panel 1: Posisi
axes[0].scatter(t_data, s_data, s=40, color='steelblue', zorder=5, label='Data (noise)')
axes[0].plot(t_fit, s_fit, 'k-', lw=1.5, label='s(t) eksak')
axes[0].set_xlabel('t (s)'); axes[0].set_ylabel('s (m)')
axes[0].set_title('Posisi s(t)'); axes[0].legend()

# Panel 2: Kecepatan
axes[1].plot(t_data, v_numpy, 'ro-', ms=5, lw=1.5, label='numpy.gradient')
axes[1].plot(t_data, v_manual, 'b^-', ms=4, lw=1.2, label='Manual beda pusat')
axes[1].plot(t_fit, v_fit, 'k-', lw=2, label='v(t) eksak')
axes[1].axhline(v_terminal, color='gray', ls=':', lw=1.2, label=f'v_terminal={v_terminal} m/s')
axes[1].set_xlabel('t (s)'); axes[1].set_ylabel('v (m/s)')
axes[1].set_title('Kecepatan v=ds/dt'); axes[1].legend(fontsize=9)

# Panel 3: Percepatan
axes[2].plot(t_data, a_numpy, 'mD-', ms=5, lw=1.5, label='a=d(v_numpy)/dt')
axes[2].plot(t_fit, a_fit, 'k-', lw=2, label='a(t) eksak')
axes[2].axhline(g, color='blue', ls=':', lw=1.2, label=f'g={g} m/s²')
axes[2].axhline(0, color='gray', ls='--', lw=1)
axes[2].set_xlabel('t (s)'); axes[2].set_ylabel('a (m/s²)')
axes[2].set_title('Percepatan a=d²s/dt²'); axes[2].legend(fontsize=9)

plt.suptitle('Kinematika Gerak Jatuh Bebas dengan Hambatan Udara', fontsize=13)
plt.tight_layout()
plt.savefig('kinematika.png', dpi=150, bbox_inches='tight')
plt.show()

# Galat RMS
rms_v = np.sqrt(np.mean((v_numpy - v_eksak)**2))
rms_a = np.sqrt(np.mean((a_numpy - a_eksak)**2))
print(f'\nGalat RMS kecepatan: {rms_v:.4f} m/s')
print(f'Galat RMS percepatan: {rms_a:.4f} m/s²')
print('(Galat sebagian besar dari noise, bukan dari metode numerik)')

# SESI 4A: Dispersi Optik Indeks Bias vs Panjang Gelombang
import numpy as np
import matplotlib.pyplot as plt

# Koefisien Sellmeier untuk kaca BK7 (crown glass)
B1, B2, B3 = 1.03961212, 0.23179234, 1.01046945
C1, C2, C3 = 6.00069867e-3, 2.00179144e-2, 1.03560653e2 # dalam µm²

def n_sellmeier(lam_um):
    """Indeks bias kaca BK7, λ dalam mikrometer"""
    lam2 = lam_um**2
    return np.sqrt(1 + B1*lam2/(lam2-C1) + B2*lam2/(lam2-C2) + B3*lam2/(lam2-C3))

# Array panjang gelombang
lamarr = np.linspace(0.40, 1.00, 200)
hopt = 1e-5
narr = n_sellmeier(lamarr)

dn_dlam = np.array([beda_pusat(n_sellmeier, l, hopt) for l in lamarr])
d2n_dlam = np.array([beda_kedua(n_sellmeier, l, hopt) for l in lamarr])

# Koefisien dispersi grup (Group Velocity Dispersion)
c_light = 3e8 # m/s
GVD = (lamarr * 1e-6 / c_light) * d2n_dlam * 1e24 # ps²/km

# Panjang gelombang Zero Dispersion (GVD=0)
sign_change = np.where(np.diff(np.sign(GVD)))[0]
lam_ZD = lamarr[sign_change[0]] if len(sign_change) > 0 else None
