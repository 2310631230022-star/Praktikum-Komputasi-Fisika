# ============================================================
# SESI 1: Implementasi Manual Trapesium & Simpson Komposit
# Integral uji: ∫ ₀ ¹ e^(-x²) dx
# ============================================================
import numpy as np
import matplotlib.pyplot as plt
from scipy import integrate

def f(x): return np.exp(-x**2)
a, b = 0.0, 1.0

# Nilai 'eksak' dari scipy
I_eksak, err_scipy = integrate.quad(f, a, b)
print(f'Nilai referensi (scipy.quad): I = {I_eksak:.15f}')

# ─── ATURAN TRAPESIUM KOMPOSIT ───────────────────────────
def trapesium(f, a, b, n):
    x = np.linspace(a, b, n+1)
    h = (b - a) / n
    return h/2 * (f(x[0]) + 2*np.sum(f(x[1:-1])) + f(x[-1]))

# ─── ATURAN SIMPSON 1/3 KOMPOSIT ─────────────────────────
def simpson13(f, a, b, n):
    if n % 2 != 0: raise ValueError('n harus genap untuk Simpson 1/3!')
    x = np.linspace(a, b, n+1)
    h = (b - a) / n
    koef = np.ones(n+1)
    koef[1:-1:2] = 4  # titik ganjil: bobot 4
    koef[2:-2:2] = 2  # titik genap interior: bobot 2
    return h/3 * np.dot(koef, f(x))

# ─── ATURAN SIMPSON 3/8 ──────────────────────────────────
def simpson38(f, a, b, n):
    if n % 3 != 0: raise ValueError('n harus kelipatan 3 untuk Simpson 3/8!')
    x = np.linspace(a, b, n+1)
    h = (b - a) / n
    koef = np.ones(n+1)
    for i in range(1, n):
        koef[i] = 3 if i % 3 != 0 else 2
    koef[0] = koef[-1] = 1
    return 3*h/8 * np.dot(koef, f(x))

# ─── TABEL PERBANDINGAN UNTUK BERBAGAI n ─────────────────
n_list = [2, 4, 8, 16, 32, 64, 128, 256]
print(f"\n{'n':>5} {'Trapesium':>16} {'|E_trap|':>12} {'Simpson 1/3':>16} {'|E_simp|':>12}")
print('-' * 70)
for n in n_list:
    I_trap = trapesium(f, a, b, n)
    I_simp = simpson13(f, a, b, n)
    e_trap = abs(I_trap - I_eksak)
    e_simp = abs(I_simp - I_eksak)
    print(f'{n:>5} {I_trap:>16.12f} {e_trap:>12.2e} {I_simp:>16.12f} {e_simp:>12.2e}')

# ─── ANALISIS ORDE KONVERGENSI: GRAFIK LOG-LOG ───────────
n_arr = np.array([4, 8, 16, 32, 64, 128, 256, 512])
h_arr = (b - a) / n_arr
e_trap = np.array([abs(trapesium(f,a,b,n) - I_eksak) for n in n_arr])
e_simp = np.array([abs(simpson13(f,a,b,n) - I_eksak) for n in n_arr])

fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(13, 5))

# Panel kiri: konvergensi
ax1.loglog(h_arr, e_trap, 'r-o', lw=2, ms=6, label='Trapesium Komposit')
ax1.loglog(h_arr, e_simp, 'b-s', lw=2, ms=6, label='Simpson 1/3 Komposit')
ax1.loglog(h_arr, 0.05*h_arr**2, 'r--', alpha=0.6, lw=1, label='O(h²)')
ax1.loglog(h_arr, 0.001*h_arr**4, 'b--', alpha=0.6, lw=1, label='O(h⁴)')
ax1.set_xlabel('h = (b−a)/n'); ax1.set_ylabel('|Galat|')
ax1.set_title('Konvergensi: Trapesium vs Simpson'); ax1.legend()

# Panel kanan: visualisasi panel
n_vis = 4
x_data = np.linspace(a, b, n_vis+1)
x_fine = np.linspace(a, b, 300)
ax2.fill_between(x_fine, f(x_fine), alpha=0.2, color='green', label='Nilai eksak')
ax2.plot(x_fine, f(x_fine), 'k-', lw=2)

for i in range(n_vis):
    xi = x_data[i:i+2]
    ax2.fill_between(xi, f(xi), alpha=0.4, color='red', step='pre')
ax2.scatter(x_data, f(x_data), s=60, color='red', zorder=5)
ax2.set_xlabel('x'); ax2.set_ylabel('f(x) = e^(-x²)')
ax2.set_title(f'Trapesium Komposit n={n_vis}'); ax2.legend()

plt.tight_layout()
plt.show()

Sesi 2: Integrasi Romberg 

# ============================================================
# SESI 2: Integrasi Romberg — Tabel Ekstrapolasi Richardson
# ============================================================
def romberg(f, a, b, max_level=8, tol=1e-12):
    '''Integrasi Romberg hingga konvergen.'''
    R = np.zeros((max_level+1, max_level+1))
    
    # Kolom 0: Trapesium dengan 2^n sub-interval
    R[0,0] = (b-a)/2 * (f(a) + f(b))
    
    for n in range(1, max_level+1):
        h = (b-a) / 2**n
        new_pts = np.arange(1, 2**n, 2)
        R[n,0] = R[n-1,0]/2 + h * np.sum(f(a + new_pts*h))
        
        # Kolom Richardson
        for m in range(1, n+1):
            R[n,m] = (4**m * R[n,m-1] - R[n-1,m-1]) / (4**m - 1)
            
        # Cek konvergensi
        if n > 0 and abs(R[n,n] - R[n-1,n-1]) < tol:
            return R[n,n], R[:n+1,:n+1], n
    return R[max_level,max_level], R, max_level

# ─── JALANKAN ROMBERG ─────────────────────────────────────
I_romberg, tabel_R, level = romberg(f, a, b, max_level=10, tol=1e-14)
print(f'=== INTEGRASI ROMBERG ===')
print(f'Hasil Romberg: {I_romberg:.15f}')
print(f'Level konvergensi: n = {level}')
print(f'Galat vs scipy: {abs(I_romberg - I_eksak):.2e}\n')

# Cetak tabel Romberg
print('Tabel Romberg R[n,m]:')
header = 'n \t' + '\t'.join([f'm={m}' for m in range(min(level+1,6))])
print(header)
for n in range(min(level+1, 7)):
    row = f'{n}\t'
    for m in range(n+1):
        row += f'{tabel_R[n,m]:.8f}\t'
    print(row)

print('\nHubungan Romberg-Simpson:')
print(f'R[2,1] = {tabel_R[2,1]:.10f} (Romberg kolom 1)')
print(f'Simpson13 n=4: {simpson13(f,a,b,4):.10f} (harus sama!)')

# SESI 3: Kuadratur Gauss-Legendre dari nol & perbandingan
# ============================================================
from numpy.polynomial.legendre import leggauss

def gauss_legendre(f, a, b, n):
    '''Kuadratur Gauss-Legendre n titik pada [a,b].'''
    xi, wi = leggauss(n)
    x_mapped = (b-a)/2 * xi + (b+a)/2
    return (b-a)/2 * np.dot(wi, f(x_mapped))

# ─── PERBANDINGAN EFISIENSI: GALAT vs EVALUASI FUNGSI ────
print('=== GAUSS-LEGENDRE vs TRAPESIUM vs SIMPSON ===')
print(f"{'n_eval':>7} {'Trapesium':>14} {'Simpson':>14} {'Gauss':>14}")
print('-' * 56)

for n_gauss in [2, 3, 4, 5, 6, 8, 10]:
    n_trap = n_gauss
    n_simp = n_gauss if n_gauss%2==0 else n_gauss+1
    
    e_t = abs(trapesium(f,a,b,n_trap) - I_eksak)
    e_s = abs(simpson13(f,a,b,n_simp) - I_eksak)
    e_g = abs(gauss_legendre(f,a,b,n_gauss) - I_eksak)
    print(f'{n_gauss+1:>7} {e_t:>14.2e} {e_s:>14.2e} {e_g:>14.2e}')

# ─── KONVERGENSI GAUSS: FUNGSI ANALITIK vs TIDAK MULUS ───
n_gauss_arr = np.arange(1, 20)
e_gauss_smooth = [abs(gauss_legendre(f,a,b,n)-I_eksak) for n in n_gauss_arr]

# Fungsi tidak mulus: |x−0.5|
g = lambda x: np.abs(x - 0.5)
I_g_eksak, _ = integrate.quad(g, 0, 1)

e_gauss_rough = [abs(gauss_legendre(g,a,b,n)-I_g_eksak) for n in n_gauss_arr]
e_trap_rough = [abs(trapesium(g,a,b,n*4)-I_g_eksak) for n in n_gauss_arr]

fig, axes = plt.subplots(1, 2, figsize=(13, 5))
axes[0].semilogy(n_gauss_arr, e_gauss_smooth, 'b-o', lw=2, ms=5, label='Gauss (f mulus e^-x²)')
axes[0].semilogy(n_gauss_arr, [abs(trapesium(f,a,b,4*n)-I_eksak) for n in n_gauss_arr], 'r-s', lw=2, label='Trapesium (4n panel)')
axes[0].set_title('Gauss: Konvergensi Eksponensial'); axes[0].legend()

axes[1].semilogy(n_gauss_arr, e_gauss_rough, 'b-o', lw=2, ms=5, label='Gauss (|x−0.5|)')
axes[1].semilogy(n_gauss_arr, e_trap_rough, 'r-s', lw=2, ms=5, label='Trapesium (4n panel)')
axes[1].set_title('Gauss: Tidak Selalu Menang!'); axes[1].legend()
plt.show()

# ============================================================
# SESI 4A: Periode Bandul Nonlinear Eksak
# ============================================================
import numpy as np
import matplotlib.pyplot as plt
from scipy import integrate

L = 1.0; g = 9.81
T0 = 2*np.pi*np.sqrt(L/g)

def periode_bandul(theta0_deg):
    theta0 = np.radians(theta0_deg)
    k = np.sin(theta0/2)
    def integrand(phi): return 1.0 / np.sqrt(1 - k**2 * np.sin(phi)**2)
    K, _ = integrate.quad(integrand, 0, np.pi/2)
    T_eks = 4 * np.sqrt(L/g) * K
    T_gauss5 = gauss_legendre(integrand, 0, np.pi/2, 5) * 4*np.sqrt(L/g)
    return T_eks, T_gauss5

print('=== PERIODE BANDUL NONLINEAR ===')
print(f"{'θ ₀ (°)':>8} {'T_eksak (s)':>14} {'T_gauss5 (s)':>14} {'T/T ₀':>8}")
for th in [10, 30, 60, 90, 120, 170]:
    T_eks, T_g5 = periode_bandul(th)
    print(f'{th:>8} {T_eks:>14.8f} {T_g5:>14.8f} {T_eks/T0:>8.4f}')

# ============================================================
# SESI 4B: Fungsi Partisi Termodinamika (Kapasitas Panas Debye)
# ============================================================
def debye_Cv(T, theta_D=343.0, N=1):
    kB = 1.0
    x_D = theta_D / T
    def integrand(x):
        with np.errstate(over='ignore', invalid='ignore'):
            ex = np.exp(x)
            result = x**4 * ex / (ex - 1)**2
        return np.where(x < 1e-10, 1.0, result)
    I, _ = integrate.quad(integrand, 1e-10, x_D)
    return 9 * N * kB * (T/theta_D)**3 * I

def debye_gauss(T, theta_D=343.0, n_pts=20):
    x_D = theta_D / T
    def integrand(x):
        with np.errstate(over='ignore', invalid='ignore'):
            ex = np.exp(np.clip(x, 0, 700))
            r = x**4 * ex / (ex - 1)**2
        return np.where(x < 1e-10, 1.0, r)
    I_gauss = gauss_legendre(integrand, 1e-10, x_D, n_pts)
    return 9 * (T/theta_D)**3 * I_gauss

T_arr = np.array([10, 50, 100, 200, 300, 400, 500])
print('\n=== KAPASITAS PANAS DEBYE (Aluminium, Θ_D = 343 K) ===')
print(f"{'T (K)':>6} {'C_V (Debye)':>14} {'C_V (Gauss-10)':>16}")
for T in T_arr:
    cv_q = debye_Cv(T)
    cv_g = debye_gauss(T, n_pts=10)
    print(f'{T:>6} {cv_q:>14.8f} {cv_g:>16.8f}')
