def regres_polinomial(x, y, m):
    '''
    Regresi polinomial orde m dari Normal Equations.
    Parameter:
    x, y : array data
    m    : orde polinomial
    Return:
    a      : koefisien [a0, a1, ..., am]
    R2     : koefisien determinasi
    SS_res : jumlah kuadrat residual
    '''
    n = len(x)
    # Bangun matriks desain (Vandermonde) X berukuran n×(m+1)
    X = np.column_stack([x**k for k in range(m+1)])
    
    # Normal Equations: (XᵀX)·a = Xᵀy
    XtX = X.T @ X
    Xty = X.T @ y
    
    # Selesaikan sistem linear
    a = np.linalg.solve(XtX, Xty)
    
    # Hitung R²
    y_pred = X @ a
    SS_res = np.sum((y - y_pred)**2)
    SS_tot = np.sum((y - y.mean())**2)
    R2     = 1 - SS_res / SS_tot
    
    return a, R2, SS_res

# ─── BANDINGKAN BERBAGAI ORDE ─────────────────────────────

# SESI 3: Hukum Pendinginan Newton — Linearisasi Eksponensial
# Model: T(t) = T∞ + (T₀−T∞)·e^(-k·t)
# Setelah transformasi: ln(T(t)−T∞) = ln(T₀−T∞) − k·t
# Bentuk linear: Y = A + B·t, di mana Y = ln(T−T∞)
#
# Data simulasi: Bola besi panas didinginkan di udara
# ============================================================
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import curve_fit

# ─── DATA PENGAMATAN ──────────────────────────────────────
# Waktu (menit) dan suhu (°C)
t_data = np.array([0, 2, 4, 6, 8, 10, 12, 15, 20, 25, 30], dtype=float)
T_data = np.array([90.0, 78.5, 68.8, 60.2, 53.4, 47.8, 43.1, 37.2, 29.6, 24.8, 22.1], dtype=float)
T_inf = 21.0 # suhu lingkungan (°C)

print('=== DATA PENDINGINAN NEWTON ===')
print(f'Suhu lingkungan T∞ = {T_inf}°C')
print(f't (menit): {t_data}')
print(f'T (°C)   : {T_data}')

# ─── LINEARISASI: Y = ln(T − T∞) ─────────────────────────
dT = T_data - T_inf # kelebihan suhu di atas T∞
Y  = np.log(dT)     # variabel transformasi

print(f'\nT − T∞ : {np.round(dT, 2)}')
print(f'ln(T−T∞): {np.round(Y, 4)}')

# ─── REGRESI LINEAR pada (t, Y) ───────────────────────────
n     = len(t_data)
x_bar = np.mean(t_data)
Y_bar = np.mean(Y)

b_lin = np.sum((t_data - x_bar)*(Y - Y_bar)) / np.sum((t_data - x_bar)**2)
a_lin = Y_bar - b_lin * x_bar

# Kembali ke parameter fisika
k_lin  = -b_lin                 # laju pendinginan [1/menit]
T0_lin = np.exp(a_lin) + T_inf  # suhu awal terfitting

Y_pred = a_lin + b_lin * t_data
R2_lin = 1 - np.sum((Y - Y_pred)**2) / np.sum((Y - Y_bar)**2)

print(f'\n=== HASIL LINEARISASI ===')
print(f'Intersep A = ln(T₀−T∞) = {a_lin:.4f}')
print(f'Kemiringan B = −k      = {b_lin:.4f}')
print(f'k (laju pendinginan)   = {k_lin:.4f} /menit')
print(f'T₀ terfitting          = {T0_lin:.2f}°C (data: {T_data[0]}°C)')
print(f'R² (pada Y=ln(T−T∞))   = {R2_lin:.6f}')

# ─── VERIFIKASI: curve_fit langsung di domain y asli ────
def model_newton(t, T0, k):
    return T_inf + (T0 - T_inf) * np.exp(-k * t)

popt, pcov = curve_fit(model_newton, t_data, T_data, p0=[90.0, 0.1], maxfev=5000)
T0_cf, k_cf = popt
perr = np.sqrt(np.diag(pcov)) # standard error

T_pred_cf = model_newton(t_data, *popt)
SS_res_cf = np.sum((T_data - T_pred_cf)**2)
SS_tot_cf = np.sum((T_data - T_data.mean())**2)
R2_cf     = 1 - SS_res_cf / SS_tot_cf

print(f'\n=== VERIFIKASI scipy.optimize.curve_fit ===')
print(f'k  = {k_cf:.4f} ± {perr[1]:.4f} /menit')
print(f'T₀ = {T0_cf:.2f} ± {perr[0]:.2f} °C')
print(f'R² (pada T asli) = {R2_cf:.6f}')
print(f'\nSelisih k: {abs(k_lin-k_cf):.4f} /menit')
print('→ Linearisasi menghasilkan koefisien sedikit berbeda karena')
print(' meminimalkan residual pada ln(T) bukan pada T!')

# ─── VISUALISASI PENDINGINAN NEWTON ──────────────────────
t_fit = np.linspace(0, 35, 300)
fig, axes = plt.subplots(1, 3, figsize=(16, 5))

# Panel 1: T vs t (domain asli)
axes[0].scatter(t_data, T_data, s=70, color='steelblue', zorder=5, label='Data')
T_pred_lin = T_inf + np.exp(a_lin + b_lin * t_fit)
axes[0].plot(t_fit, T_pred_lin, 'r--', lw=2, label=f'Linearisasi (k={k_lin:.3f})')
axes[0].plot(t_fit, model_newton(t_fit, *popt), 'g-', lw=2, label=f'curve_fit (k={k_cf:.3f})')
axes[0].axhline(T_inf, color='gray', ls=':', lw=1, label=f'T∞={T_inf}°C')
axes[0].set_xlabel('Waktu (menit)'); axes[0].set_ylabel('Suhu (°C)')
axes[0].set_title('Hukum Pendinginan Newton'); axes[0].legend(fontsize=9)

# Panel 2: ln(T−T∞) vs t (domain linear)
axes[1].scatter(t_data, Y, s=70, color='darkorange', zorder=5, label='ln(T−T∞)')
axes[1].plot(t_fit, a_lin + b_lin*t_fit, 'r-', lw=2, label=f'Regresi linear\nR²={R2_lin:.4f}')
axes[1].set_xlabel('Waktu (menit)'); axes[1].set_ylabel('ln(T − T∞)')
axes[1].set_title('Domain Terliniearisasi'); axes[1].legend(fontsize=9)

# Panel 3: Residual perbandingan
res_lin = T_data - (T_inf + np.exp(a_lin + b_lin * t_data))
res_cf  = T_data - model_newton(t_data, *popt)

plt.tight_layout()
plt.savefig('pendinginan_newton.png', dpi=150, bbox_inches='tight')

print(f'\nWaktu paruh pendinginan: t½ = ln(2)/k = {t_half:.2f} menit')

def beda_5titik(f, x, h):
    # Formula 5-titik untuk turunan pertama - galat O(h^4)
    return (-f(x+2*h) + 8*f(x+h) - 8*f(x-h) + f(x-2*h)) / (12*h)

def richardson(f, x, h):
    # Richardson Extrapolation: O(h^2) -> O(h^4)
    D_h = beda_pusat(f, x, h)
    D_h2 = beda_pusat(f, x, h/2)
    return (4*D_h2 - D_h) / 3

def f(x): return np.exp(-x**2)
a, b = 0.0, 1.0

# Nilai 'eksak' dari scipy
I_eksak, err_scipy = integrate.quad(f, a, b)
print(f'Nilai referensi (scipy.quad): I = {I_eksak:.15f}')

# ─── ATURAN TRAPESIUM KOMPOSIT ───────────────────────────
def trapesium(f, a, b, n):
    x = np.linspace(a, b, n+1)
    h = (b - a) / n
    return h/2 * (f(x[0]) + 2*np.sum(f(x[1:-1])) + f(x[-1]))

# ─── ATURAN SIMPSON 1/3 KOMPOSIT ─────────────────────────
def simpson13(f, a, b, n):
    if n % 2 != 0: raise ValueError('n harus genap untuk Simpson 1/3!')
    x = np.linspace(a, b, n+1)
    h = (b - a) / n
    koef = np.ones(n+1)
    koef[1:-1:2] = 4  # titik ganjil: bobot 4
    koef[2:-2:2] = 2  # titik genap interior: bobot 2
    return h/3 * np.dot(koef, f(x))

# ─── ATURAN SIMPSON 3/8 ──────────────────────────────────
def simpson38(f, a, b, n):
    if n % 3 != 0: raise ValueError('n harus kelipatan 3 untuk Simpson 3/8!')
    x = np.linspace(a, b, n+1)
    h = (b - a) / n
    koef = np.ones(n+1)
    for i in range(1, n):
        koef[i] = 3 if i % 3 != 0 else 2
    koef[0] = koef[-1] = 1
    return 3*h/8 * np.dot(koef, f(x))

# SESI 2: Integrasi Romberg — Tabel Ekstrapolasi Richardson
# ============================================================
def romberg(f, a, b, max_level=8, tol=1e-12):
    '''Integrasi Romberg hingga konvergen.'''
    R = np.zeros((max_level+1, max_level+1))
    
    # Kolom 0: Trapesium dengan 2^n sub-interval
    R[0,0] = (b-a)/2 * (f(a) + f(b))
    
    for n in range(1, max_level+1):
        h = (b-a) / 2**n
        new_pts = np.arange(1, 2**n, 2)
        R[n,0] = R[n-1,0]/2 + h * np.sum(f(a + new_pts*h))
        
        # Kolom Richardson
        for m in range(1, n+1):
            R[n,m] = (4**m * R[n,m-1] - R[n-1,m-1]) / (4**m - 1)
            
        # Cek konvergensi
        if n > 0 and abs(R[n,n] - R[n-1,n-1]) < tol:
            return R[n,n], R[:n+1,:n+1], n
    return R[max_level,max_level], R, max_level

from numpy.polynomial.legendre import leggauss

def gauss_legendre(f, a, b, n):

#SESI 2: Eliminasi Gauss dengan Partial Pivoting
import numpy as np

def eliminasi_gauss(A, b, verbose=False):
    """Menyelesaikan Ax=b dengan Eliminasi Gauss + Partial Pivoting.
    Parameters:
    A (ndarray): Matriks koefisien n x n
    b (ndarray): Vektor konstanta (n,)
    verbose : Tampilkan detail setiap langkah
    Returns:
    x (ndarray): Vektor solusi
    """
    n = len(b)
    
    # Salin ke matriks augmented jangan ubah input asli!
    Aug = np.hstack([A.copy().astype(float), b.copy().reshape(-1,1).astype(float)])
    
    if verbose:
        print('Augmented awal [A|b]:')
        print(np.round(Aug, 4)); print()
        
    # FASE 1: FORWARD ELIMINATION dengan Partial Pivoting
    for k in range(n):
        # Partial pivoting: cari baris dengan nilai absolut terbesar
        idx_pivot = np.argmax(np.abs(Aug[k:, k])) + k
        
        if idx_pivot != k:
            Aug[[k, idx_pivot]] = Aug[[idx_pivot, k]]
            if verbose:
                print(f'[Pivot] Tukar baris {k} <-> {idx_pivot}')
                
        # Cek singularitas
        if abs(Aug[k, k]) < 1e-12:
            raise ValueError(f'Matriks singular pada pivot k={k}!')
            
        # Eliminasi elemen di bawah pivot
        for i in range(k+1, n):
            m = Aug[i, k] / Aug[k, k] # Faktor pengali
            Aug[i, :] -= m * Aug[k, :] # Operasi baris
            
        if verbose:
            print(f'Setelah eliminasi kolom {k}:')
            print(np.round(Aug, 4)); print()
            
    # FASE 2: BACK SUBSTITUTION (Substitusi Mundur)
    x = np.zeros(n)
    for i in range(n-1, -1, -1): # dari baris bawah ke atas
        sigma = np.dot(Aug[i, i+1:n], x[i+1:n])
        x[i] = (Aug[i, n] - sigma) / Aug[i, i]
        
    return x

#SESI 3: Invers Matriks (Gauss-Jordan) dan Determinan
import numpy as np

def gauss_jordan_invers(A):
    """Hitung A^(-1) menggunakan metode Gauss-Jordan."""
    n = A.shape[0]
    Aug = np.hstack([A.copy().astype(float), np.eye(n)])
    
    for k in range(n):
        # Partial pivoting
        p = np.argmax(np.abs(Aug[k:, k])) + k
        Aug[[k, p]] = Aug[[p, k]]
        
        if abs(Aug[k, k]) < 1e-12:
            raise ValueError('Matriks singular tidak ada invers!')
            
        # Normalisasi baris pivot
        Aug[k, :] /= Aug[k, k]
        
        # Eliminasi ke ATAS dan ke BAWAH (Gauss-Jordan)
        for i in range(n):
            if i != k:
                Aug[i, :] -= Aug[i, k] * Aug[k, :]
                
    return Aug[:, n:] # Invers ada di bagian kanan

def hitung_determinan(A):
    """Hitung det(A) menggunakan Eliminasi Gauss.
    det = sign * produk elemen diagonal segitiga atas."""
    n = A.shape[0]
    M = A.copy().astype(float)
    sgn = 1 # tanda berubah tiap pertukaran baris
    
    for k in range(n):
        p = np.argmax(np.abs(M[k:, k])) + k
        if p != k:
            M[[k, p]] = M[[p, k]]
            sgn *= -1
            
        if abs(M[k, k]) < 1e-12:
            return 0.0
            
        for i in range(k+1, n):
            M[i, :] -= (M[i, k]/M[k, k]) * M[k, :]
            
    return sgn * np.prod(np.diag(M))
